class PerfilAdm extends Component {

    constructor() {
        super();
        this.state = {
            nome: "",
            email: "",
            senha: "",
            usuarios: [],
            admin: "",
            listaUsuarios: ""
        }
        this.BuscarAdm = this.BuscarAdm.bind(this);
        this.camposet = this.camposet.bind(this);

    }
        // Execulta o componente criado
        componentDidMount (){
            this.BuscarAdm();
        }

        camposet(event) { // botão de definição dos setState para função buscar admim abaixo
            this.setState({
                nome: event.target.value,
                email: event.target.value,
                senha: event.target.value
            })
        }

        // método get
        async BuscarAdm() { 
            await fetch('http://localhost:5000/api/usuario')
            // Executa 
            .then(Response =>Response.json())
            .then(data => this.setState({listaUsuarios: data}))
            .catch(error => console.error(error))

            console.log(this.state.listaUsuarios)
                       
            this.setState({nome: this.state.listaUsuarios.nome})
            this.setState({email: this.state.listaUsuarios.email})
            this.setState({senha: this.state.listaUsuarios.senha})         
            }  
        }

    render() {
            
        return (
          
            <div className = "CorpoTodo">
                <Acessibilidade/>
                {/* <MenuS/> */}
                
                    <div class="conteudoCentral">
                            {/* <MenuEs/> */}
                        <div className="infoUser">
                            <section className="conteudo">

                                <h1> Perfil do Administrador </h1>
                                <div class="separador">
                                    <hr size="1" color="#D3D3D3"/>
                                </div>

                                <div id="info">

                                    <form id="box-form">
                                        <div id="form1">
                                            <label>Nome:</label>
                                            <input type="text" name="nome" value={this.state.admin.nome} className="nomeUser" placeholder=" Nome Completo" />

                                            <label>E-mail:</label>
                                            <input type="text" name="email" value={this.state.admin.email} className="cEmailUser"  placeholder="Seu Email" />

                                            <label>Senha:</label>
                                            <input type="text" name="senha" value={this.state.admin.senha} className="confSenha" placeholder="Sua senha" />
                                        </div>
                                    </form>
                                </div>
                            </section>
                            </div>
                        </div>
                        <Rodape/>
            </div>
        );
        }
export default PerfilAdm;
